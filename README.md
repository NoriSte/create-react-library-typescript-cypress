## Install

```bash
npm install --save create-react-library-typescript-cypress
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'create-react-library-typescript-cypress'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
